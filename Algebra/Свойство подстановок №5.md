$$
\begin{align}
\sigma = \tau_1 \ldots \tau_k \\
\tau^{-1}_i = \tau_i \text{ , так как транспозиция} \\
\sigma^{-1} = \tau_k \ldots \tau_1 \\
I(\sigma) \equiv k \equiv I(\sigma^{-1}) \ (mod \ 2)\\
\end{align}
$$
